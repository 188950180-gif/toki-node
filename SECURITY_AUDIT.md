**建议：**
```rust
// 添加消息验证
pub fn validate_message(msg: &NetworkMessage) -> Result<()> {
    // 大小限制
    if msg.size() > MAX_MESSAGE_SIZE {
        return Err(Error::MessageTooLarge);
    }

    // 签名验证
    msg.verify_signature()?;

    // 时间戳验证（防止重放）
    if msg.timestamp < now() - MAX_MESSAGE_AGE {
        return Err(Error::MessageExpired);
    }

    Ok(())
}
```
